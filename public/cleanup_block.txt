    var _docsCache = null;
    async function loadDocuments() {
        if (_docsCache) return _docsCache;
        if (!window.CloudDB || !window.CloudDB.loadSheet) {
            console.warn('CloudDB not available, documents list empty');
            return [];
        }
        try {
            var docs = await window.CloudDB.loadSheet('Worker_Documents');
            _docsCache = Array.isArray(docs) ? docs : [];
            return _docsCache;
        } catch(e) {
            console.warn('Failed to load Worker_Documents:', e);
            return [];
        }
    }

    function getDocName(docId, docs) {
        var d = (docs || []).find(function(x){ return String(x.DocumentID) === String(docId); });
        return d ? (d.DocumentNameEN || d.DocumentName || d.Name || ('Doc #' + docId)) : ('Doc #' + docId);
    }

    function parseMaybeArray(v) {
        if (Array.isArray(v)) return v;
        if (typeof v === 'string') {
            try { var j = JSON.parse(v); return Array.isArray(j) ? j : []; } catch(e) { return []; }
        }
        return [];
    }

    function parseMaybeObject(v) {
        if (v && typeof v === 'object' && !Array.isArray(v)) return v;
        if (typeof v === 'string') {
            try { var j = JSON.parse(v); return (j && typeof j === 'object' && !Array.isArray(j)) ? j : {}; } catch(e) { return {}; }
        }
        return {};
    }

    function fileExt(name){ var m=/\.([a-z0-9]+)$/i.exec(name||''); return m?m[1].toLowerCase():''; }
    function isImageExt(e){ return /^(jpe?g|png|gif|webp)$/i.test(e); }
    function isPdfExt(e){ return e==='pdf'; }
    function escHtml(s){ return String(s).replace(/[&<>"']/g, function(c){ return {'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]; }); }

    window.spDeleteUploadedDoc = async function(docId, fileIdx) {
        if (!window.currentWorker) return;
        if (!confirm('Delete this uploaded file?')) return;
        var uploaded = parseMaybeObject(window.currentWorker.UploadedWorkerDocs);
        var arr = uploaded[docId] || [];
        var removed = arr.splice(fileIdx, 1)[0];
        uploaded[docId] = arr;
        window.currentWorker.UploadedWorkerDocs = uploaded;

        // Try to delete from storage
        if (removed && removed.path && window.CloudStorage && window.CloudStorage.deleteFile) {
            try { await window.CloudStorage.deleteFile(removed.path); } catch(e) { console.warn('Delete failed:', e); }
        }

        await window.CloudDB.saveRow('Workers', window.currentWorker);
        window.loadWorkerUploadList();
    };

    function attachDragDropHandlers() {
        var slots = document.querySelectorAll('#workerUploadList .sp-doc-slot');
        slots.forEach(function(slot) {
            var dropzone = slot.querySelector('.sp-dropzone');
            var input = slot.querySelector('.sp-file-input');
            var docId = slot.getAttribute('data-docid');
            var docName = slot.getAttribute('data-docname');

            function handleFiles(files) {
                if (!files || !files.length) return;
                spUploadFilesForDoc(docId, docName, Array.from(files));
            }

            if (input) {
                input.addEventListener('change', function(e){ handleFiles(e.target.files); });
            }

            if (dropzone) {
                ['dragenter','dragover'].forEach(function(ev){
                    dropzone.addEventListener(ev, function(e){
                        e.preventDefault(); e.stopPropagation();
                        slot.style.borderColor = '#667eea';
                        slot.style.background = '#eef2ff';
                    });
                });
                ['dragleave','drop'].forEach(function(ev){
                    dropzone.addEventListener(ev, function(e){
                        e.preventDefault(); e.stopPropagation();
                        slot.style.borderColor = '#cbd5e1';
                        slot.style.background = '#f9fafb';
                    });
                });
                dropzone.addEventListener('drop', function(e){
                    e.preventDefault(); e.stopPropagation();
                    handleFiles(e.dataTransfer.files);
                });
            }
        });
    }

    async function spUploadFilesForDoc(docId, docName, files) {
        if (!window.currentWorker) { alert('?? Please select a worker first!'); return; }
        if (!window.CloudStorage || !window.CloudStorage.uploadFile) { alert('?? Upload not available.'); return; }

        // Filter oversized files
        var MAX = 10 * 1024 * 1024;
        var tooBig = files.filter(function(f){ return f.size > MAX; });
        if (tooBig.length) {
            alert('? ' + tooBig.length + ' file(s) exceed 10 MB limit:\n' + tooBig.map(function(f){ return '• ' + f.name; }).join('\n'));
            files = files.filter(function(f){ return f.size <= MAX; });
            if (!files.length) return;
        }

        var worker = window.currentWorker;
        var supplier = window.currentSupplier || {};
        var sc = String(supplier.SupplierCODE || supplier.SupplierCode || '').toUpperCase();
        var wid = worker.WorkerID;
        var wid4 = String(wid).padStart(4, '0');
        var fullName = worker.WFullName || worker.WorkerFirstName || 'UNNAMED';

        // Always prefer existing WorkerFolder (set by portal's upload) for consistency
        var folder = worker.WorkerFolder;
        if (!folder) {
            if (typeof window.workerFolderKey === 'function') folder = window.workerFolderKey(sc, wid, fullName);
            else folder = sc + ' (' + wid4 + ') ' + fullName;
        }

        var uploaded = parseMaybeObject(worker.UploadedWorkerDocs);
        var existingArr = uploaded[docId] || [];
        var existingCount = existingArr.length;

        var uploadedOk = [];
        var failed = [];

        if (typeof window.showUploadOverlay === 'function') {
            window.showUploadOverlay('?? Uploading ' + docName, '1 of ' + files.length);
        }

        for (var i = 0; i < files.length; i++) {
            var file = files[i];
            var copyNum = existingCount + i + 1;
            var ext = file.name.substring(file.name.lastIndexOf('.')) || '';
            // Filename pattern: {DocumentNameEN}[N]-{SupCode-WorkerID4d}.{ext}
            var versionSuffix = (copyNum === 1) ? '' : String(copyNum);
            var fileName = docName + versionSuffix + '-' + sc + '-' + wid4 + ext;
            var path = 'workers/' + folder + '/' + fileName;

            if (typeof window.showUploadOverlay === 'function') {
                window.showUploadOverlay('?? Uploading ' + docName, (i+1) + ' of ' + files.length + ': ' + file.name);
            }

            try {
                var result = await window.CloudStorage.uploadFile(path, file);
                if (result && result.success) {
                    uploadedOk.push({
                        fileName: fileName,
                        displayName: docName + (versionSuffix ? ' (' + versionSuffix + ')' : ''),
                        baseName: docName,
                        version: copyNum,
                        fileUrl: result.url,
                        url: result.url,
                        path: path,
                        docId: docId,
                        uploadedAt: new Date().toISOString()
                    });
                } else {
                    failed.push({ fileName: file.name, error: (result && result.error) || 'Unknown error' });
                }
            } catch(e) {
                failed.push({ fileName: file.name, error: e.message || String(e) });
            }
        }

        // Update worker
        if (uploadedOk.length) {
            existingArr = existingArr.concat(uploadedOk);
            uploaded[docId] = existingArr;
            worker.UploadedWorkerDocs = uploaded;
            // Make sure WorkerFolder is persisted
            if (!worker.WorkerFolder) worker.WorkerFolder = folder;
            try {
                await window.CloudDB.saveRow('Workers', worker);
            } catch(e) {
                console.warn('Worker save failed:', e);
            }
        }

        if (typeof window.hideUploadOverlay === 'function') {
            window.hideUploadOverlay(uploadedOk.length > 0, uploadedOk.length + ' uploaded');
        }

        // Refresh UI
        window.loadWorkerUploadList();

        if (failed.length) {
            alert('?? ' + uploadedOk.length + ' uploaded, ' + failed.length + ' failed:\n\n' +
                failed.map(function(f){ return '• ' + f.fileName + ': ' + f.error; }).join('\n'));
        }
    }

    // Override loadWorkerUploadList — supplier-driven
    window.loadWorkerUploadList = async function() {
        var container = document.getElementById('workerUploadList');
        if (!container) return;

        if (!window.currentWorker) {
            container.innerHTML = '<p style="padding:20px;text-align:center;color:#999;">?? Please select a worker first!</p>';
            return;
        }

        container.innerHTML = '<p style="padding:20px;text-align:center;color:#999;">? Loading required documents...</p>';

        var supplier = window.currentSupplier || {};
        var requiredDocs = parseMaybeArray(supplier.RequiredWorkerDocs);

        var docs = await loadDocuments();

        // If no required docs configured on supplier, show all active docs as fallback
        var slotList;
        if (requiredDocs.length) {
            slotList = requiredDocs.map(function(docId){
                return { docId: docId, docName: getDocName(docId, docs) };
            });
        } else if (docs.length) {
            slotList = docs.filter(function(d){ return d.Active !== false; })
                .map(function(d){ return { docId: String(d.DocumentID), docName: (d.DocumentNameEN || d.DocumentName || ('Doc #' + d.DocumentID)) }; });
            if (slotList.length) {
                container.innerHTML = '<div style="background:#fef3c7;border:1px solid #fbbf24;border-radius:8px;padding:10px 14px;margin-bottom:12px;color:#92400e;font-size:12px;">?? No required docs configured on supplier — showing all active document types.</div>' +
                    '<div id="spSlotsWrap"></div>';
            }
        } else {
            container.innerHTML = '<p style="padding:20px;text-align:center;color:#999;">?? No documents configured.</p>';
            return;
        }

        if (!slotList || !slotList.length) {
            container.innerHTML = '<p style="padding:20px;text-align:center;color:#999;">?? No required documents for this supplier.</p>';
            return;
        }

        var uploaded = parseMaybeObject(window.currentWorker.UploadedWorkerDocs);

        // Header info
        var worker = window.currentWorker;
        var sc = String((supplier.SupplierCODE || supplier.SupplierCode || '')).toUpperCase();
        var wid4 = String(worker.WorkerID).padStart(4, '0');
        var wname = worker.WFullName || worker.WorkerFirstName || 'Worker';
        var header = '<div style="background:linear-gradient(135deg,#eef2ff,#e0e7ff);border-radius:10px;padding:10px 14px;margin-bottom:14px;font-size:13px;color:#3730a3;">' +
            '<strong>?? ' + escHtml(sc + '-' + wid4) + ' — ' + escHtml(wname) + '</strong>' +
            '<span style="margin-left:10px;color:#6b7280;font-size:11px;">' + slotList.length + ' required document type' + (slotList.length !== 1 ? 's' : '') + '</span>' +
        '</div>';

        var slotsHtml = '';
        slotList.forEach(function(item){
            slotsHtml += renderDocSlot(item.docId, item.docName, uploaded[item.docId] || []);
        });

        // If we already rendered the warning wrapper, inject slots inside it
        var wrap = document.getElementById('spSlotsWrap');
        if (wrap) {
            wrap.innerHTML = header + slotsHtml;
        } else {
            container.innerHTML = header + slotsHtml;
        }

        attachDragDropHandlers();
    };

            s.onerror = function(){ reject(new Error('Failed to load pdf-lib')); };
            document.head.appendChild(s);
        });
        return _pdfLibPromise;
    }

    function makeDraggable(box, handle){
        var sx=0, sy=0, bl=0, bt=0, dragging=false;
        handle.addEventListener('mousedown', function(e){
            if (e.target.closest('button')) return;
            dragging=true; sx=e.clientX; sy=e.clientY;
            var r=box.getBoundingClientRect();
            bl=r.left; bt=r.top;
            box.style.top=bt+'px'; box.style.left=bl+'px';
            box.style.width=r.width+'px'; box.style.height=r.height+'px';
            document.body.style.userSelect='none';
            e.preventDefault();
        });
        document.addEventListener('mousemove', function(e){
            if (!dragging) return;
            box.style.left=(bl+e.clientX-sx)+'px';
            box.style.top=(bt+e.clientY-sy)+'px';
        });
        document.addEventListener('mouseup', function(){ if(dragging){dragging=false; document.body.style.userSelect='';} });
    }
    function makeResizable(box, handle){
        var sx=0, sy=0, sw=0, sh=0, resizing=false;
        handle.addEventListener('mousedown', function(e){
            resizing=true; sx=e.clientX; sy=e.clientY;
            var r=box.getBoundingClientRect();
            sw=r.width; sh=r.height;
            box.style.top=r.top+'px'; box.style.left=r.left+'px';
            document.body.style.userSelect='none';
            e.preventDefault(); e.stopPropagation();
        });
        document.addEventListener('mousemove', function(e){
            if (!resizing) return;
            box.style.width=Math.max(500, sw+e.clientX-sx)+'px';
            box.style.height=Math.max(400, sh+e.clientY-sy)+'px';
        });
        document.addEventListener('mouseup', function(){ if(resizing){resizing=false; document.body.style.userSelect='';} });
    }

    async function downloadOne(url, name){
        try {
            var resp=await fetch(url);
            var blob=await resp.blob();
            var link=document.createElement('a');
            link.href=URL.createObjectURL(blob); link.download=name;
            document.body.appendChild(link); link.click();
            setTimeout(function(){ document.body.removeChild(link); URL.revokeObjectURL(link.href); },500);
        } catch(e){ window.open(url,'_blank'); }
    }

    window.spfmPrintSelected = async function(ev){
        var sel=_state.files.filter(function(f){ return _state.selected[f.path||f.name]; });
        if (!sel.length){ alert('?? No files selected.'); return; }
        var btn=ev&&ev.target; var origText=btn?btn.textContent:'';
        if (btn){ btn.textContent='? Building...'; btn.disabled=true; }
        try {
            var PDFLib=await ensurePdfLib();
            var outPdf=await PDFLib.PDFDocument.create();
            var HEADER_H = 24;
            var headerBg = PDFLib.rgb(0.4, 0.31, 0.64);
            var headerFg = PDFLib.rgb(1, 1, 1);
            function stamp(page, name, idx, total, op) {
                var sz=page.getSize(); op=(op===undefined)?1:op;
                page.drawRectangle({x:0,y:sz.height-HEADER_H,width:sz.width,height:HEADER_H,color:headerBg,opacity:op});
                var dn=name.length>80?name.substring(0,77)+'...':name;
                page.drawText(dn,{x:10,y:sz.height-16,size:10,color:headerFg,opacity:op});
                page.drawText((idx+1)+' / '+total,{x:sz.width-50,y:sz.height-16,size:10,color:headerFg,opacity:op});
            }
            for (var i=0; i<sel.length; i++){
                var f=sel[i], url=f.url||f.fileUrl||'', ext=fileExt(f.name);
                if (btn) btn.textContent='? '+(i+1)+'/'+sel.length;
                try {
                    var resp=await fetch(url); var buf=await resp.arrayBuffer();
                    if (isPdfExt(ext)){
                        try {
                            var src=await PDFLib.PDFDocument.load(buf,{ignoreEncryption:true,throwOnInvalidObject:false});
                            var pages=await outPdf.copyPages(src,src.getPageIndices());
                            pages.forEach(function(p,pi){
                                outPdf.addPage(p);
                                stamp(p, f.name+(pages.length>1?' (p.'+(pi+1)+'/'+pages.length+')':''), i, sel.length, 0.85);
                            });
                        } catch(pe){
                            var ep=outPdf.addPage([595,842]); stamp(ep,f.name+' (error)',i,sel.length);
                            ep.drawText('PDF error: '+pe.message.substring(0,60),{x:40,y:400,size:12});
                        }
                    } else if (ext==='jpg'||ext==='jpeg'){
                        var img=await outPdf.embedJpg(buf);
                        var pg=outPdf.addPage([img.width,img.height+HEADER_H]);
                        pg.drawImage(img,{x:0,y:0,width:img.width,height:img.height});
                        stamp(pg,f.name,i,sel.length);
                    } else if (ext==='png'){
                        var imgP=await outPdf.embedPng(buf);
                        var pgP=outPdf.addPage([imgP.width,imgP.height+HEADER_H]);
                        pgP.drawImage(imgP,{x:0,y:0,width:imgP.width,height:imgP.height});
                        stamp(pgP,f.name,i,sel.length);
                    } else {
                        var np=outPdf.addPage([595,842]); stamp(np,f.name,i,sel.length);
                        np.drawText('Unsupported: '+ext,{x:40,y:400,size:12});
                    }
                } catch(e){
                    var eep=outPdf.addPage([595,842]); stamp(eep,f.name+' (error)',i,sel.length);
                    eep.drawText('Error: '+e.message.substring(0,60),{x:40,y:400,size:12});
                }
            }
            var bytes=await outPdf.save();
            var blob=new Blob([bytes],{type:'application/pdf'});
            var w=window.open(URL.createObjectURL(blob),'_blank');
            if (w) w.addEventListener('load', function(){ setTimeout(function(){try{w.print();}catch(e){}},400); });
        } catch(e){
            alert('? Print failed: '+(e.message||e));
        } finally {
            if (btn){ btn.textContent=origText; btn.disabled=false; }
        }
    };

    window.spOpenAllFiles = async function() {
        if (!window.currentWorker) { alert('?? No worker selected.'); return; }
        if (!window.CloudStorage) { alert('?? CloudStorage not available.'); return; }

        var folder = window.currentWorker.WorkerFolder;
        if (!folder) { alert('?? Worker has no folder yet. Upload files first.'); return; }

        var modal = buildModal();
        modal.style.display = 'block';

        var folderPath = 'workers/' + folder;
        var sc = String((window.currentSupplier && (window.currentSupplier.SupplierCODE || window.currentSupplier.SupplierCode)) || '').toUpperCase();
        var wid4 = String(window.currentWorker.WorkerID).padStart(4, '0');
        var wname = window.currentWorker.WFullName || window.currentWorker.WorkerFirstName || '';

        document.getElementById('spfmSub').textContent = 'Loading...';
        document.getElementById('spfmViewer').innerHTML = '<div class="sfm-placeholder"><div class="ico">?</div><div>Loading files...</div></div>';

        var result;
        // Try CloudStorage.listFiles if available
        if (window.CloudStorage.listFiles) {
            try { result = await window.CloudStorage.listFiles(folderPath); } catch(e) { result = { success: false, error: e.message }; }
        } else {
            // REST API fallback
            try {
                var BUCKET = 'mgsd2026.firebasestorage.app';
                var listUrl = 'https://firebasestorage.googleapis.com/v0/b/' + BUCKET + '/o?prefix=' + encodeURIComponent(folderPath + '/');
                var resp = await fetch(listUrl);
                if (!resp.ok) { result = { success:false, error:'List failed: '+resp.status, files:[] }; }
                else {
                    var data = await resp.json();
                    var files = (data.items||[]).map(function(item){
                        var fullPath = item.name;
                        var justName = fullPath.split('/').pop();
                        var downloadUrl = 'https://firebasestorage.googleapis.com/v0/b/' + BUCKET + '/o/' + encodeURIComponent(fullPath) + '?alt=media';
                        return { name: justName, path: fullPath, url: downloadUrl, fileUrl: downloadUrl };
                    });
                    result = { success:true, files:files };
                }
            } catch(e) { result = { success:false, error:e.message, files:[] }; }
        }

        if (!result || !result.success) {
            document.getElementById('spfmViewer').innerHTML = '<div class="sfm-placeholder"><div class="ico">?</div><div>' + ((result && result.error) || 'Unknown error') + '</div></div>';
            return;
        }

        var files = (result.files || []).slice();
        if (!files.length) {
            document.getElementById('spfmViewer').innerHTML = '<div class="sfm-placeholder"><div class="ico">??</div><div>No files yet.</div></div>';
            document.getElementById('spfmSub').textContent = folderPath + ' — empty';
            return;
        }

        files.sort(function(a,b){
            function rank(n){ var e=fileExt(n); if (isImageExt(e)) return 0; if (isPdfExt(e)) return 1; return 2; }
            var ra=rank(a.name), rb=rank(b.name);
            if (ra!==rb) return ra-rb;
            return a.name.localeCompare(b.name);
        });

        _state.files=files; _state.idx=0; _state.selected={};
        _state.view='single'; _state.zoom=1;

        document.getElementById('spfmSub').textContent = sc + '-' + wid4 + (wname?' — '+wname:'') + ' — ' + files.length + ' file' + (files.length!==1?'s':'');
        document.getElementById('spfmSelectAll').checked = false;
        showPage(0);
    };
